slgfx Tutorial

This tutorial shows some simple use cases for how to use slgfx. It assumes at least a basic working knowledge of HTML and Javascript.

Full documentation can be found here.

To start, import the bundle.js or bundle.dev.js file:

<script type="text/javascript" src="./dist/bundle.dev.js"></script>

Note: All classes in the bundled js files are exported to the SL namespace for use in the browser. You will notice all calls to classes below are prefixed with "SL.".

Then create a div to hold the Screen:

<div id="screenDiv"></div>

Before we do anything else, it is a good idea to load up the image(s) that we will be working with later in the tutorial. We can use the ImageLoader class to tell us when the image has loaded, then we can safely call any other code we want.

Now we can begin seeting up our screen and elements.

To setup a basic screen:

var screen = new SL.Screen({ targetDiv:document.getElementById("{screenId}"), width:128, height:96 }); screen.initialize(); var gfxLayer = screen.createLayer("GfxLayer"); screen.render();

Create an ImageRenderer, an ImageElement, and add it to the canvas context:

var screen = new SL.Screen({ targetDiv:document.getElementById("{screenId}"), width:128, height:96 }); screen.initialize(); var gfxLayer = screen.createLayer("GfxLayer"); screen.render(); var canvasContext = gfxLayer.getCanvasContextWrapper(); var imageRenderer = new SL.ImageRenderer(1,1); var element = new SL.ImageElement({ screenContext: screen, canvasContextWrapper: canvasContext, image: imagesHash.gnome.image, sourceX:0, sourceY:0, sourceWidth:32, sourceHeight:32, width:32, height:32, scaleX:2, scaleY:2, imageRenderer:imageRenderer }); gfxLayer.addElement(element);

Let's move the element, reset its position, and repeat:

var screen = new SL.Screen({ targetDiv:document.getElementById("{screenId}"), width:128, height:96 }); screen.initialize(); var gfxLayer = screen.createLayer("GfxLayer"); screen.render(); var canvasContext = gfxLayer.getCanvasContextWrapper(); var imageRenderer = new SL.ImageRenderer(1,1); var element = new SL.ImageElement({ screenContext: screen, canvasContextWrapper: canvasContext, image: imagesHash.gnome.image, sourceX:0, sourceY:0, sourceWidth:32, sourceHeight:32, width:32, height:32, scaleX:2, scaleY:2, imageRenderer:imageRenderer }); gfxLayer.addElement(element); function resetMovement(element) { element.setX(0); element.setY(0); element.moveTo(32, 32, 5000, resetMovement.bind(null, element)); } resetMovement(element);

Instead of a static image, let's create a sprite instead:

var screen = new SL.Screen({ targetDiv:document.getElementById("{screenId}"), width:128, height:96 }); screen.initialize(); var gfxLayer = screen.createLayer("GfxLayer"); screen.render(); var canvasContext = gfxLayer.getCanvasContextWrapper(); var imageRenderer = new SL.ImageRenderer(1,1); var element = new SL.ImageSprite({ screenContext: screen, canvasContextWrapper: canvasContext, image:imagesHash.gnome.image, width:32, height:32, scaleX:2, scaleY:2, imageRenderer:imageRenderer, frames: [ new SL.ImageSpriteFrame({duration:350, sourceX:0, sourceY:0, sourceWidth:32, sourceHeight:32}), new SL.ImageSpriteFrame({duration:350, sourceX:64, sourceY:0, sourceWidth:32, sourceHeight:32}) ] }) gfxLayer.addElement(element);

We can flip the element:

var screen = new SL.Screen({ targetDiv:document.getElementById("{screenId}"), width:128, height:96 }); screen.initialize(); var gfxLayer = screen.createLayer("GfxLayer"); screen.render(); var canvasContext = gfxLayer.getCanvasContextWrapper(); var imageRenderer = new SL.ImageRenderer(1,1); var element = new SL.ImageSprite({ screenContext: screen, canvasContextWrapper: canvasContext, image:imagesHash.gnome.image, width:32, height:32, scaleX:2, scaleY:2, imageRenderer:imageRenderer, frames: [ new SL.ImageSpriteFrame({duration:350, sourceX:0, sourceY:0, sourceWidth:32, sourceHeight:32}), new SL.ImageSpriteFrame({duration:350, sourceX:64, sourceY:0, sourceWidth:32, sourceHeight:32}) ] }) gfxLayer.addElement(element); element.setHorizontallyFlipped(true);

We can leverage events on the screen to make it rotate over time:

var screen = new SL.Screen({ targetDiv:document.getElementById("{screenId}"), width:128, height:96 }); screen.initialize(); var gfxLayer = screen.createLayer("GfxLayer"); screen.render(); var canvasContext = gfxLayer.getCanvasContextWrapper(); var imageRenderer = new SL.ImageRenderer(1,1); var element = new SL.ImageSprite({ screenContext: screen, canvasContextWrapper: canvasContext, image:imagesHash.gnome.image, width:32, height:32, scaleX:2, scaleY:2, imageRenderer:imageRenderer, frames: [ new SL.ImageSpriteFrame({duration:350, sourceX:0, sourceY:0, sourceWidth:32, sourceHeight:32}), new SL.ImageSpriteFrame({duration:350, sourceX:64, sourceY:0, sourceWidth:32, sourceHeight:32}) ] }) gfxLayer.addElement(element); screen.on(SL.EventType.BEFORE_RENDER, function(element, e) { var degrees = (e.time / 10) % 360; element.setRotation( SL.Utils.degreesToRadians(degrees) ); }.bind(null, element));

Let's make the element shake when clicked:

var screen = new SL.Screen({ targetDiv:document.getElementById("{screenId}"), width:128, height:96 }); screen.initialize(); var gfxLayer = screen.createLayer("GfxLayer"); screen.render(); var canvasContext = gfxLayer.getCanvasContextWrapper(); var imageRenderer = new SL.ImageRenderer(1,1); var element = new SL.ImageSprite({ screenContext: screen, canvasContextWrapper: canvasContext, image:imagesHash.gnome.image, width:32, height:32, scaleX:2, scaleY:2, imageRenderer:imageRenderer, frames: [ new SL.ImageSpriteFrame({duration:350, sourceX:0, sourceY:0, sourceWidth:32, sourceHeight:32}), new SL.ImageSpriteFrame({duration:350, sourceX:64, sourceY:0, sourceWidth:32, sourceHeight:32}) ] }) gfxLayer.addElement(element); var currentlyShaking = false; element.on(SL.EventType.MOUSE_UP_ON_ELEMENT, function(e) { if (currentlyShaking) return; currentlyShaking = true; this.shake(20, 2, 50, 0, 3000, function() {currentlyShaking = false;}); }.bind(element));
